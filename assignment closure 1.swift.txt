import Foundation

let rightJustify = {
    (initialString : String) //There should not be space before :
    in

    var stringLength = initialString.count
    var result = "" //After variable declaration, have a blank line. It will be easy to read
    for _ in 0...(70 - stringLength) {
        result += "\u{00a0}"
    }
    result += initialString
    print(result)
    
}
print(rightJustify("vinayaka"))
